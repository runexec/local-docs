Simple websocket example

Setting up a connection to a websocket is a really simple thing to do with Closure.
Here is a little example class that sets up a connection, listen for new messages and dispatch an event which can be listened to from anywhere you might wish.

goog.require('goog.net.WebSocket');
goog.require('goog.events.EventHandler');
goog.require('goog.events.EventTarget');
 
/**
  * @constructor
  */
app.socket = function(){
  this.ws_  = new goog.net.WebSocket();
  this.url_ = 'wss://someurl:1234';
  this.eh_  = new goog.events.EventHandler();
 
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.OPENED,  this.onOpen, false, this);
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.MESSAGE, this.onMessage, false, this);
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.CLOSED,  this.onClosed, false, this);
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.ERROR,   this.onError, false, this);
}
goog.inherits(app.socket, goog.events.EventTarget);
 
app.socket.prototype.init = function(){
  this.ws_.open(this.url_);
}
 
app.socket.prototype.onOpen = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['type'],
    'message_type' : 'open'
  });
}
 
app.socket.prototype.onMessage = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['message'],
    'message_type' : 'message'
  });
}
 
app.socket.prototype.onClosed = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['type'],
    'message_type' : 'error'
  });
}
 
app.socket.prototype.onError = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['type'],
    'message_type' : 'error'
  });
}
goog.require('goog.net.WebSocket');
goog.require('goog.events.EventHandler');
goog.require('goog.events.EventTarget');

/**
  * @constructor
  */
app.socket = function(){
  this.ws_  = new goog.net.WebSocket();
  this.url_ = 'wss://someurl:1234';
  this.eh_  = new goog.events.EventHandler();

  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.OPENED,  this.onOpen, false, this);
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.MESSAGE, this.onMessage, false, this);
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.CLOSED,  this.onClosed, false, this);
  this.eh_.listen(this.ws_, goog.net.WebSocket.EventType.ERROR,   this.onError, false, this);
}
goog.inherits(app.socket, goog.events.EventTarget);

app.socket.prototype.init = function(){
  this.ws_.open(this.url_);
}

app.socket.prototype.onOpen = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['type'],
    'message_type' : 'open'
  });
}

app.socket.prototype.onMessage = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['message'],
    'message_type' : 'message'
  });
}

app.socket.prototype.onClosed = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['type'],
    'message_type' : 'error'
  });
}

app.socket.prototype.onError = function(e){
  this.dispatchEvent({
    'type'         : 'PUSHMESSAGE',
    'message'      : e['type'],
    'message_type' : 'error'
  });
}
